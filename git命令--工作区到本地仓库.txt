1，常见设置
git config --global  user.name '名称'	---设置用户名
git config --global  user.email '邮箱'	---设置邮箱

2，创建本地仓库
git init

3，本地文件与缓存区的操作
git add 文件名		---将本地文件加入缓存区
git reset HEAD		---将本地文件从缓存区撤回

4，缓存区与本地仓库的操作
git commit -m 'log描述'		---将缓存区文件加入本地仓库
git diff HEAD --文件名		---查看缓存区与本地仓库中，当前文件内容的不同
git rm 文件名				---从本地仓库移除某个文件
git checkout --文件名		---从本地仓库拉取某个文件
git ls-files				---查看本地仓库中的所有文件

5，其他操作
git status						---查看提交情况
git log							---查看提交记录的log内容
git log -行数 pretty=oneline	---提交记录的log以单行显示
cat 文件名						---查看某个文件中的内容
git relog						---查看以前的所有操作记录

6，版本操作
git reset --hard HEAD^			---回退一个版本，个数根据^符号的个数而定
git reset --hard HEAD~2			---回退两个版本，个数根据~后的数据而定
git reset --hard git版本的Id	---跳转到指定的版本
